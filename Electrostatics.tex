\section*{Coulomb's Law \& Electric Field}
\begin{flushleft}
    \textbf{Source charges} $q$ — responsible for creating electric field/force

    \textbf{Test charge} $Q$ — charge used to measure the exisiting electric field/force

    \textbf{Principal of Superposition} — interaction between two charges is independant of other charges
\end{flushleft}

\subsection*{Coulomb's Law}
    \begin{equation}
        \vec{F} = \cc \frac{qQ}{\rcurs^2}\hrcurs
    \end{equation}
    Where $\vec{\brcurs} = \vec{r} - \vec{r'}$ is the \textbf{separation vector}, $\vec{r}$ \& $\vec{r'}$ are \textbf{position vectors} for test charge and source charge respectively\\
    \textbf{Permitivity of free space} $\epsilon_0 \equiv 8.85\times 10^{-12}\frac{C^2}{N\cdot m^2}$

\subsection*{The Electric Field}
    Force due to multiple charges,
    \begin{gather*}
        \vec{F} = \vec{F}_{total} = \vec{F}_1 + \vec{F}_2 = \cc \left(\frac{q_1Q}{\rcurs^2_1}\hrcurs_1 +\frac{q_2Q}{\rcurs^2_2}\hrcurs_2 + \dots\right)\\
        \vec{F} = \frac{Q}{4\pi\epsilon_0}\left[\frac{q_1}{\rcurs^2_1}\hrcurs_1 +\frac{q_2}{\rcurs^2_2}\hrcurs_2 + \dots\right]
    \end{gather*}
    $\vec{E} \equiv$ electric field of source charges.
    \begin{gather}
        \vec{E} = \frac{1}{4\pi\epsilon_0}\left[\frac{q_1}{\rcurs^2_1}\hrcurs_1 +\frac{q_2}{\rcurs^2_2}\hrcurs_2 + \dots\right] = \cc \sum_{i=1}^{n}\frac{q_i}{\rcurs^2_i}\hrcurs_i
    \end{gather}
    then,
    \begin{gather}
        \vec{F} = Q\vec{E} (\vec{r})
    \end{gather}
    $\vec{E} (\vec{r}) \equiv$ force per unit charge on a test charge at point P
    \newpage
    \hrule
    \vspace{6px}
    \subitem \textbf{Example 2.1} |
        Find the electric field a distance $z$ above the midpoint between two equal charges $q$, a distance $d$ apart.
    \subitem \textbf{Solution:}
    \begin{gather*}
        \text{Point charges | }\pointEF\\
        \text{Let }\vec{E}_1 \text{ be the electric field of the left charge and $\vec{E}_2$ be the field of the right charge.}\\
        \text{By symmetry, the horizontal components cancel, so find the vertical components and add.}\\
        E_{1z}: q = q, \quad \rcurs = \sqrt{z^2 + \left(d/2\right)^2} \quad
        \vec{\brcurs} = (d/2)\hat{x} + z\hat{z} \quad \hrcurs = \frac{\vec{\brcurs}}{\rcurs} = \frac{z}{\sqrt{z^2 + (d/2)^2}}\hat{z}\\
        \vec{E}_1z = \cc \frac{q}{\sqrt{z^2 + \left(d/2\right)^2}^2}\frac{z}{\sqrt{z^2 + (d/2)^2}}\hat{z} = \cc \frac{qz}{\left(z^2 + \left(d/2\right)\right)^{3/2}}\hat{z}\\
        \vec{E_{1z}} = \vec{E_{2z}} \longrightarrow \vec{E_{1z}} + \vec{E_{2z}} = \boxed{\cc \frac{2qz}{\left(z^2 + \left(d/2\right)\right)^{3/2}}\hat{z}}
    \end{gather*}

    \vspace{5px}\hrule

\subsection*{Continuous Charge Distributions}
    If the charge is continuously distributed over some region, the sum in (Eq. 2) becomes an integral.
    \begin{equation}
        \vec{E} (\vec{r}) = \cc\int \frac{1}{\rcurs^2}\hrcurs dq
    \end{equation}
    \subitem Electric field of a \textbf{line charge}, $dq \rightarrow \lambda dl'$ where $dl'$ is an element of length along the line,
    \begin{equation}
        \vec{E} (\vec{r}) = \cc \int \frac{\lambda (\vec{r})}{\rcurs^2}\hrcurs dl'
    \end{equation}
    \subitem Electric field for a \textbf{surface charge}, $dq \rightarrow \sigma da'$ where $da'$ is an element of area on the surface,
    \begin{equation}
        \vec{E} (\vec{r}) = \cc \int \frac{\sigma (\vec{r})}{\rcurs^2}\hrcurs da'
    \end{equation}
    \subitem Electric field for a \textbf{volume charge}, $dq \rightarrow \rho dl'$ where $d\tau'$ is an element of volume,
    \begin{equation}
        \vec{E} (\vec{r}) = \cc \int \frac{\rho (\vec{r})}{\rcurs^2}\hrcurs d\tau'
    \end{equation}
    \newpage
    \hrule
    \vspace{5px}
    \subitem \textbf{Example 2.2} | Find the electric field a distance $z$ above the midpoint of a straigh line segment of length $2L$ that carries a uniform line charge $\lambda$
    \begin{equation*}
        \vec{E} (\vec{r}) = \cc \int \frac{\lambda (\vec{r})}{\rcurs^2}\hrcurs dl'
    \end{equation*}
    \begin{gather*}
        \text{Horizontal components are going to cancel}\\
        dq = \lambda dl', \text{ $dl'$ is $dx$ along x-axis} \quad \vec{r} = z\hat{z} \text{ (test position vector)} \quad \vec{r'} = x\hat{x} \text{ (source position vector)}\\
        \vec{\brcurs} = z\hat{z} - x\hat{x} \quad \rcurs = \sqrt{z^2 + x^2} \quad \hrcurs = \frac{z\hat{z} - x\hat{x}}{\sqrt{z^2 + x^2}}\\
        \vec{E} (\vec{r}) = \cc \int_{-L}^{L} \frac{\lambda}{\left(\sqrt{z^2 + x^2}\right)^2}\frac{z\hat{z} - x\hat{x}}{\sqrt{z^2 + x^2}}dx
    \end{gather*}
