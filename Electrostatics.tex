\section*{Coulomb's Law \& Electric Field}
\begin{flushleft}
    \textbf{Source charges} $q$ — responsible for creating electric field/force

    \textbf{Test charge} $Q$ — charge used to measure the exisiting electric field/force

    \textbf{Principal of Superposition} — interaction between two charges is independant of other charges
\end{flushleft}

\subsection*{Coulomb's Law}
    \begin{equation}
        \vec{F} = \cc \frac{qQ}{\rcurs^2}\hrcurs
    \end{equation}
    Where $\vec{\brcurs} = \vec{r} - \vec{r'}$ is the \textbf{separation vector}, $\vec{r}$ \& $\vec{r'}$ are \textbf{position vectors} for test charge and source charge respectively\\
    \textbf{Permitivity of free space} $\epsilon_0 \equiv 8.85\times 10^{-12}\frac{C^2}{N\cdot m^2}$

\subsection*{The Electric Field}
    Force due to multiple charges,
    \begin{gather*}
        \vec{F} = \vec{F}_{total} = \vec{F}_1 + \vec{F}_2 = \cc \left(\frac{q_1Q}{\rcurs^2_1}\hrcurs_1 +\frac{q_2Q}{\rcurs^2_2}\hrcurs_2 + \dots\right)\\
        \vec{F} = \frac{Q}{4\pi\epsilon_0}\left[\frac{q_1}{\rcurs^2_1}\hrcurs_1 +\frac{q_2}{\rcurs^2_2}\hrcurs_2 + \dots\right]
    \end{gather*}
    $\vec{E} \equiv$ electric field of source charges.
    \begin{gather}
        \vec{E} = \frac{1}{4\pi\epsilon_0}\left[\frac{q_1}{\rcurs^2_1}\hrcurs_1 +\frac{q_2}{\rcurs^2_2}\hrcurs_2 + \dots\right] = \cc \sum_{i=1}^{n}\frac{q_i}{\rcurs^2_i}\hrcurs_i
    \end{gather}
    then,
    \begin{gather}
        \vec{F} = Q\vec{E} (\vec{r})
    \end{gather}
    $\vec{E} (\vec{r}) \equiv$ force per unit charge on a test charge at point P
